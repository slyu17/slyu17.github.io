Arduino Code

                //variables for three pins representing red, gree and blue
                int R = 9;
                int G = 10;
                int B = 11;
                // variable for reading the pushbutton status
                int buttonState = 0;         
                
                void setup() {
                  // initialize the pushbutton pin as an input:
                  pinMode(8,INPUT);
                  // initialize the LED pins as outputs:
                  pinMode(R, OUTPUT);
                  pinMode(G, OUTPUT);
                  pinMode(B, OUTPUT);
                }
                
                void loop() {
                  // read the state of the pushbutton value:
                  buttonState = digitalRead(8);
                  // check if the pushbutton is pressed. If it is, the buttonState is HIGH.
                  if (buttonState == HIGH) {
                    // turn LEDs on with initialized color:
                    analogWrite(R, 227);
                    analogWrite(G, 113);
                    analogWrite(B, 67);
                
                    // fade in from min to max in increments of 1 points on red:
                    for (int i = 0 ; i <= 255; i++) {
                    // sets the value (range from 0 to 255):
                    analogWrite(R, i);
                    // wait for 10 milliseconds to see the dimming effect
                    delay(10);
                    }
                    // wait for 1 second
                    delay(1000);
                
                    // fade in from min to max in increments of 1 points on blue:
                    for (int i = 0 ; i <= 255; i++) {
                    // sets the value (range from 0 to 255):
                    analogWrite(B, i);
                    // wait for 10 milliseconds to see the dimming effect
                    delay(10);
                    }
                    // wait for 1 second
                    delay(1000);
                
                    // fade in from max to min in increments of 1 points on green:
                    for (int i = 255 ; i >= 0; i--) {
                    // sets the value (range from 0 to 255):
                    analogWrite(G, i);
                    // wait for 10 milliseconds to see the dimming effect
                    delay(10);
                    }
                    // wait for 1 second
                    delay(1000);
                    
                  }
                  //if the buttonState is not HIGH
                  else {
                    // turn LED off:
                    analogWrite(R, 0);
                    analogWrite(G, 0);
                    analogWrite(B, 0);
                  }
                
                }
